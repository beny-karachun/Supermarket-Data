אינטגרציה מהירה למפתחים

דוגמאות קוד לשימוש מהיר ביישומים שלך 
// Fetch products from the Supermarket Data API
const query = 'קוטג';
fetch(`http://localhost:3000/api/v1/products?query=${encodeURIComponent(query)}`)
  .then(response => response.json())
  .then(data => {
    console.log(`Found ${data.count} items matching "${query}":`, data.data);
  })
  .catch(err => console.error('Error fetching prices:', err));
# Python 3 code to search products
import requests
import urllib.parse

query = 'קוטג'
url = f'http://localhost:3000/api/v1/products?query={urllib.parse.quote(query)}'

response = requests.get(url)
if response.status_code == 200:
    data = response.json()
    print(f"Found {data['count']} items:")
    for item in data['data']:
        print(f"- {item['name']} ({item['barcode']})")
# Search for products by name or barcode
curl -X GET "http://localhost:3000/api/v1/products?query=%D7%A7%D7%95%D7%98%D7%92'" \
     -H "Accept: application/json"
